function abrirYT(url) {

  var isMobile = /Android|iPhone|iPad|iPod/i.test(navigator.userAgent);

  if (isMobile) {

    var appUrl = url

      .replace('https://www.youtube.com/watch', 'youtube://www.youtube.com/watch')

      .replace('https://youtu.be/', 'youtube://youtu.be/')

      .replace('https://www.youtube.com/results', 'youtube://www.youtube.com/results');

    var t = setTimeout(function(){ window.location.href = url; }, 1500);

    window.addEventListener('blur', function(){ clearTimeout(t); }, { once: true });

    window.location.href = appUrl;

  } else {

    window.open(url, '_blank', 'noopener');

  }

}

function toggleStep(el) {

  var body = el.nextElementSibling;

  var isOpen = body.classList.contains('open');

  el.classList.toggle('open', !isOpen);

  body.classList.toggle('open', !isOpen);

  updateProgress();

}


function updateProgress() {

  var steps = document.querySelectorAll('.step-title');

  var opened = document.querySelectorAll('.step-title.open').length;

  var pct = Math.round((opened / steps.length) * 100);

  document.getElementById('prog-bar').style.width = pct + '%';

  var labels = ['Comenzando…','En camino…','Avanzando…','¡Casi terminamos!','¡Completado!'];

  var idx = Math.min(Math.floor(opened / steps.length * (labels.length - 1)), labels.length - 1);

  document.getElementById('progress-label').textContent = labels[idx];

}

}
